window.processCohortData = (cohorts) => {
  let paintCohorts = '';
  for (i = 0; i < cohorts.length; i++) {
    paintCohorts += '<option value="' + i + '">' + cohorts[i].id + '</option>';
  }
  selectCohorts.innerHTML = paintCohorts;
}
